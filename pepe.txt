local SoundId = "rbxassetid://5032492305"

-- Crear un objeto de sonido
local sound = Instance.new("Sound")
sound.SoundId = SoundId
sound.Parent = workspace

-- Reproducir el sonido de forma infinita
sound.Looped = true
sound:Play()

-- Detectar cuando el jugador tiene 0 de vida
local function checkPlayerHealth(character)
    local humanoid = character:WaitForChild("HumanoidRootPart")
    humanoid.Died:Connect(function()
        sound:Stop()
        sound:Destroy()
    end)
end

-- Esperar hasta que el jugador exista
game.Players.PlayerAdded:Connect(function(player)
    player.CharacterAdded:Connect(function(character)
        checkPlayerHealth(character)
    end)
    if player.Character then
        checkPlayerHealth(player.Character)
    end
end)

local Players = game:GetService("Players")

local function crearPelo(player)
    if player.Character and player.Character:FindFirstChild("Head") then
        local pelo = Instance.new("Part")
        pelo.Name = "Pelo"
        pelo.Parent = player.Character

        local mesh = Instance.new("SpecialMesh")
        mesh.MeshId = "rbxassetid://6963196833"
        mesh.Parent = pelo

        pelo.BrickColor = BrickColor.new("White")
        pelo.Material = Enum.Material.Neon

        local cabeza = player.Character.Head
        pelo.Position = cabeza.Position
        pelo.CanCollide = false

        local soldadura = Instance.new("Weld")
        soldadura.Part0 = cabeza
        soldadura.Part1 = pelo
        soldadura.C0 = CFrame.new()
        soldadura.Parent = cabeza

        local cabezaJuego = game.Workspace.Head
        pelo.Anchored = true
        pelo.CFrame = cabezaJuego.CFrame

        local soldaduraJuego = Instance.new("Weld")
        soldaduraJuego.Part0 = cabezaJuego
        soldaduraJuego.Part1 = pelo
        soldaduraJuego.C0 = CFrame.new()
        soldaduraJuego.Parent = cabezaJuego

        local luz = Instance.new("PointLight")
        luz.Color = Color3.new(1, 1, 0) -- Color amarillo brillante
        luz.Range = 10 -- Alcance de la luz
        luz.Parent = pelo

        local materialNeon = Instance.new("SpecialMesh")
        materialNeon.MeshType = Enum.MeshType.FileMesh
        materialNeon.MeshId = "http://www.roblox.com/asset/?id=642310715"
        materialNeon.Parent = pelo
    end
end

game.Players.PlayerAdded:Connect(crearPelo)

for _, player in ipairs(game.Players:GetPlayers()) do
    crearPelo(player)
end
